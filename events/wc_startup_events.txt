# Adds class traits at the start of the game
character_event = {
	id = WCSUP.5
	
	hide_window = yes
	is_triggered_only = yes
	
	ai = yes
	min_age = 16
	only_capable = yes
	prisoner = no
	
	trigger = {
		is_untouchable_trigger = no

		is_lowborn = yes
		is_patrician = no
		
		OR = {
			is_ruler = no
			is_theocracy = yes
			tier = BARON
		}
		
		is_save_game = no
		has_class_trait_trigger = no
		potential_class_trigger = yes
	}
	immediate = {
		ai_add_random_class_effect = yes
	}
}

# Lothar's joke
character_event = {
	id = WCSUP.6
	
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_save_game = no
		character = 1008	# Anduin Lothar
	}
	
	immediate = {
		if = {
			limit = {
				has_game_rule = {
					name = sir_anduin_lothar
					value = anduin_n_lothar_d
				}
			}
			dynasty = 108
		}
		if = {
			limit = {
				has_game_rule = {
					name = sir_anduin_lothar
					value = anduin_n_lothar_n
				}
			}
			set_name = "Anduin Lothar"
		}
		if = {
			limit = {
				has_game_rule = {
					name = sir_anduin_lothar
					value = anduin_n_lothar_n_arathi_d
				}
			}
			set_name = "Anduin Lothar"
			dynasty = 16110
		}
	}
}

# Sets succ_evil_elective
character_event = {
	id = WCSUP.20
	
	hide_window = yes
	is_triggered_only = yes
	
	only_playable = yes
	
	trigger = {
		is_save_game = no
		any_demesne_title = {
			NOT = { has_law = succ_evil_elective }
			is_law_potential = succ_evil_elective
		}
	}
	
	immediate = {
		any_demesne_title = {
			limit = {
				NOT = { has_law = succ_evil_elective }
				is_law_potential = succ_evil_elective
			}
			add_law = { law = succ_evil_elective cooldown = no opinion_effect = no }
		}
	}
}

# Sets undead_province
character_event = {
	id = WCSUP.30
	
	hide_window = yes
	is_triggered_only = yes
	
	only_playable = yes
	
	trigger = {
		is_save_game = no
		any_demesne_province = {
			NOT = { has_character_modifier = undead_province }
			OR = {
				culture = scourge
				culture = kvaldir
			}
		}
	}
	
	immediate = {
		any_demesne_province = {
			limit = {
				OR = {
					culture = scourge
					culture = kvaldir
				}
			}
			add_province_modifier = { name = undead_province duration = -1 }
		}
	}
}
